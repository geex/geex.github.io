72      /**
73       * Adds a scene to the window, with a given parent and window title.
74       *
75       * @param window The window to add the scene to
76       * @param parent The parent to attach to the scene
77       * @param title  The title to give the window
78       */
79      public static void addScene(final Stage window, final Parent parent, final String title) {
80          Scene scene = new Scene(parent, SCENE_BACKGROUND);
81          window.setTitle(title);
82          scene.getStylesheets().add("/application/css/style.css");
83          window.setScene(scene);
84      }

86      /**

88       *
89       * @param window The window to style
90       */
91      public static void styleWindow(final Stage window) {
92          window.getIcons().addAll(
93                  ICONS.stream()
94                          .map(WindowFactory::getImage)
95                          .collect(toList())
96          );
97  
98          window.setMinWidth(FRAME_WIDTH);
99          window.setMinHeight(FRAME_HEIGHT);
100         window.setMaximized(true);
101     }
