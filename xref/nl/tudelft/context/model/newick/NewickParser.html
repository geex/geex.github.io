15  public final class NewickParser extends Parser<Newick> {
16      /**
17       * The node factory.
18       */
19      private NodeParser nodeParser = new NodeParser();
20      /**
21       * The vertical distances between nodes.
22       */
23      public static final int ROW_HEIGHT = 25;
24      /**
25       * The weights are scaled with this factor to make relative weights noticeable.
26       */
27      public static final double WEIGHT_SCALE = 5e4;
28      /**
29       * Child nodes are at least MIN_WEIGHT pixels moved from their parent.
30       */
31      public static final int MIN_WEIGHT = 30;

33      /**
34       * Empty constructor for creating an empty TreeParser.
35       */
36      public NewickParser() {
37          super();
38      }

40      @Override
41      protected Newick parse(final BufferedReader... readerList) {
42          BufferedReader reader = readerList[0];
43          TreeParser tp =
44                  new TreeParser(reader);
45  
46          Tree nwkTree = tp.tokenize("");
47          Node root = nodeParser.getNode(nwkTree.getRoot());
48  
49          Newick newick = new Newick();
50          getOffspring(nwkTree.getRoot(), root, newick, 0);
51          newick.setRoot(root);
52  
53          return newick;
54      }
55  
56      /**
57       * Recursive call that creates nodes and edges for the tree.
58       *
59       * @param node   the node, read by the newick parser
60       * @param parent the parent node to add this node to as a child
61       * @param newick   the tree to add the nodes and edges to
62       * @param row    the current row (depth) of the node
63       * @return the new row (depth) of the next node
64       */
65      public int getOffspring(final TreeNode node, final Node parent, final Newick newick, final int row) {
66          newick.addVertex(parent);
67  
68          int ret = row;
69  
70          int addRow = 1;
71          for (int i = 0; i < node.numberLeaves; i += 1) {
72              TreeNode child = node.getChild(i);
73              if (child != null) {
74                  addRow = 0;
75                  Node n = createNode(child, parent, ret);
76                  ret = getOffspring(child, n, newick, ret);
77                  parent.addChild(n);
78                  if (i > 0) {
79                      addDummy(parent, n, newick);
80                  } else {
81                      newick.addEdge(parent, n);
82                  }
83              }
84          }
85  
86          return ret + addRow;
87      }
88  
89      /**
90       * Creation of the node.
91       *
92       * @param child  the node, read by the newick parser
93       * @param parent the parent node to add the child to
94       * @param row    the current row (depth) of the node
95       * @return the node as a Node
96       */
97      public Node createNode(final TreeNode child, final Node parent, final int row) {
98          Node n = nodeParser.getNode(child);
99          n.setParent(parent);
100         double x = parent.translateXProperty().doubleValue() + MIN_WEIGHT + WEIGHT_SCALE * n.getWeight();
101         n.setTranslateX(x);
102         n.setTranslateY(row * ROW_HEIGHT);
103 
104         return n;
105     }
106 
107     /**
108      * Adds a dummy node, so that we can use orthogonal lines.
109      *
110      * @param parent the parent of the node
111      * @param n      the node
112      * @param newick   the tree to add the dummy to
113      */
114     public void addDummy(final Node parent, final Node n, final Newick newick) {
115         Node dummy = new Node("", 0);
116         dummy.setTranslateX(parent.translateXProperty().doubleValue());
117         dummy.setTranslateY(n.translateYProperty().doubleValue());
118         newick.addVertex(dummy);
119         newick.addEdge(parent, dummy);
120         newick.addEdge(dummy, n);
121     }
