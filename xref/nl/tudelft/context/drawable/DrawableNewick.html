39      /**
40       * Create a wrapper around the Newick to draw the Newick tree.
41       *
42       * @param newick The Newick tree to draw
43       */
44      public DrawableNewick(final Newick newick) {
45          this.newick = newick;
46  
47          newick.vertexSet().stream()
48                  .forEach(this::addVertex);
49  
50          DummyNode dummy = new DummyNode();
51          addVertex(dummy);
52          addEdge(dummy, newick.getRoot());
53          newick.getRoot().setTranslateX(START_INDENT);
54  
55          newick.getRoot().getChildren()
56                  .forEach(this::constructTree);
57      }
58  
59      /**
60       * Create a visual tree structure for the node and its children.
61       *
62       * @param node The node
63       */
64      public void constructTree(final AbstractNode node) {
65          addEdge(node.getParent(), node);
66          translateNode(node);
67          if (node.getChildren().isEmpty()) {
68              currentRow++;
69          }
70          node.getChildren()
71                  .forEach(this::constructTree);
72      }
73  
74      /**
75       * Translates the positions of the node and its children.
76       *
77       * @param node The root node to start with
78       */
79      public void translateNode(final AbstractNode node) {
80          node.translate(MIN_WEIGHT, WEIGHT_SCALE, currentRow * ROW_HEIGHT);
81      }
