120     /**
121      * Set a new main view and push it on the view stack.
122      *
123      * @param on             Controller to stack this view on
124      * @param viewController Controller containing JavaFX root
125      */
126     public void setView(final ViewController on, final ViewController viewController) {
127 
128         viewList.remove(viewList.indexOf(on) + 1, viewList.size());
129         viewList.add(viewController);
130         view.getChildren().retainAll(viewList.stream().map(ViewController::getRoot).collect(Collectors.toList()));
131         view.getChildren().add(viewController.getRoot());
132 
133         activateView();
134 
135     }

137     /**
138      * Creates a stream of visible view controllers.
139      *
140      * @return A stream of visible view controllers
141      */
142     private Stream<ViewController> getVisibleStream() {
143         return viewList.stream().filter(viewController -> viewController.getVisibilityProperty().getValue());
144     }

146     /**
147      * Set the previous view as view.
148      */
149     public final void previousView() {
150 
151         getVisibleStream()
152                 .skip(1)
153                 .reduce((previous, current) -> current)
154                 .ifPresent(viewController -> viewController.setVisibility(false));
155 
156         activateView();
157 
158     }

160     /**
161      * Go the a certain view.
162      *
163      * @param viewController View to go to
164      */
165     public void toView(final ViewController viewController) {
166 
167         int index = viewList.indexOf(viewController) + 1;
168         viewList.stream()
169                 .skip(index)
170                 .forEach(vc -> vc.setVisibility(false));
171         viewList.stream()
172                 .limit(index)
173                 .forEach(vc -> vc.setVisibility(true));
174 
175         activateView();
176 
177     }

190     /**

192      */
193     public final void activateView() {
194         Optional<ViewController> nextTopView = topView();
195         previousTopView.ifPresent(previousView -> previousView.setActivated(false));
196         nextTopView.ifPresent(nextView -> nextView.setActivated(true));
197         previousTopView = nextTopView;
198     }
